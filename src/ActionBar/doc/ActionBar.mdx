# Action Bar

The `ActionBar` component is made up of multiple `Actions` in the form of clickable icons. The action bar can be expanded to view actions with descriptive text or made smaller to view with just icons.

#### Import Syntax

```js
import ActionBar, {
  Action,
  ActionGroup,
  BottomActionGroup
} from 'calcite-react/ActionBar'
```

## Controlled ActionBar

<Playground style={{height: '400px', background: "#f3f3f3"}}>
  {() => {    
      class ActionBarStory extends React.Component {
        constructor(props) {
          super(props);
          this.state = {
            collapsed: false,
            activeAction: 'ADD_ITEM'
          };

          this.onToggleCollapse = this.onToggleCollapse.bind(this);
          this.actionSelected = this.actionSelected.bind(this);
        }

        onToggleCollapse() {
          this.setState({collapsed: !this.state.collapsed})
        }

        actionSelected(activeAction) {
          this.setState({ activeAction });
        };

        render() {
          return (
            <ActionBar
              collapsed={this.state.collapsed}
              onToggleCollapse={this.onToggleCollapse}
            >
              <Action
                onClick={() => this.actionSelected('ADD_ITEM')}
                active={this.state.activeAction === 'ADD_ITEM'}
                icon={<PlusIcon />}
              >
                Add Item
              </Action>
              <Action
                onClick={() => this.actionSelected('SAVE')}
                active={this.state.activeAction === 'SAVE'}
                icon={<SaveIcon />}
              >
                Save
              </Action>
              <Action
                onClick={() => this.actionSelected('LAYER_LIST')}
                active={this.state.activeAction === 'LAYER_LIST'}
                icon={<LayersIcon />}
              >
                Layer List
              </Action>
            </ActionBar>
          );
        }
      }

      return <ActionBarStory />;
    }}
</Playground>
